643/** \brief Convert \a arg to int
644 *
645 * This function accepts multiple bases, depending on the prefix of the arg.
646 *
647 * Basically it comes down to this (all resolve to 42 decimal)
648 *
649 * | prefix | base | example |
650 * |:------:| ----:| --------- |
651 * | [none] | 10 | 042 |
652 * | % | 2 | %101010 |
653 * | & | 8 | &52 |
654 * | $ | 16 | $2a |
655 * | 0[bB] | 2 | 0b%101010 |
656 * | 0[xX] | 16 | 0x2a |
657 *
658 * \param[in] arg string containing a possible integer literal
659 * \param[out] return_value \a arg convert to int
660 *
661 * \return 0 on success, -1 on failure
662 *
663 * \note on error, \a return_value is set to `INT_MIN`
664 */
665static int arg_to_int(const char *arg, int *return_value)
666{
667 char *tailptr;
668 int counter = 0;
669 int base = 10;
670
671 /* set value for error conditions, keeps compiler happy */
672 *return_value = INT_MIN(-2147483647 -1);
673
674 if (arg == NULL((void*)0) || *arg == '\0') {
675 return -1;
676 }
677 /* determine base */
678 switch (*arg) {
679 case '%':
680 base = 2;
681 arg++;
682 break;
683 case '&':
684 base = 8;
685 arg++;
686 break;
687 case '$':
688 arg++;
689 base = 16;
690 break;
691 case '0':
692 arg++;
693 if (*arg == 'b' || *arg == 'B') {
694 base = 2;
695 arg++;
696 } else if (*arg == 'x' || *arg == 'X') {
697 base = 16;
698 arg++;
699 }
700 break;
701 default:
702 break; /* base is already 10 */
703 }
704
705
706 *return_value = (int)strtol(arg, &tailptr, base);
707
708 if (ioutil_errno(IOUTIL_ERRNO_ERANGE4)) {
709 return -1;
710 }
711
712 /* Only whitespace is allowed after the last valid character. */
713 if (!util_check_null_string(tailptr)) {
714 while (isspace((int)tailptr[counter])) {
715 counter++;
716 }
717 tailptr += counter;
718 if (*tailptr != 0) {
719 return -1;
720 }
721 }
722 return 0;
723}

3721 * write info block
3722 */

3724 fputc(infoBlock[n], outf);
