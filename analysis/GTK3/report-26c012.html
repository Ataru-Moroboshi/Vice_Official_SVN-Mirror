420 fsimage->error_info.map[sectors] = CBMDOS_FDC_ERR_OK;
421 if (util_fpwrite(fsimage->fd, &fsimage->error_info.map[sectors], 1, offset) < 0) {
422 log_error(fsimage_dxx_log,
423 "Error writing T:%u S:%u error info to disk image.",
424 dadr->track, dadr->sector);
425 }
426 }
427
428 /* Make sure the stream is visible to other readers. */
429 fflush(fsimage->fd);
430 return 0;
431}
